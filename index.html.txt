        <h2>**AR City Quest**</h2>
        <p>1. Point camera at table/room<br>2. Tap green box to start story<br>3. Walk to find treasures/NPCs<br>4. Share score!</p>
        <button onclick="startQuest()">Start Quest</button>
        <button onclick="shareScore()">Share Score</button>

    <div id="score">Score: <span id="scoreValue">0</span></div>

    <script>
        let score = 0;
        const scoreEl = document.getElementById('scoreValue');

        // Start interactive quest
        function startQuest() {
            document.querySelector('a-box').setAttribute('scale', '0.3 0.3 0.3');
            document.querySelector('#npc').setAttribute('visible', true);
            setTimeout(() => document.querySelector('#reward1').setAttribute('visible', true), 3000);
            updateScore(10);
        }

        function updateScore(points) {
            score += points;
            scoreEl.textContent = score;
            // Persist & simple gamification
            localStorage.setItem('arQuestScore', score);
        }

        // Click listeners for interactivity
        document.querySelector('a-scene').addEventListener('click', (e) => {
            if (e.detail.cursorEl && e.detail.cursorEl.components['cursor-listener']) {
                updateScore(5); // Reward interaction
            }
            if (e.target.id === 'reward1') {
                e.target.emit('reveal');
                updateScore(20);
            }
        });

        // Viral sharing
        function shareScore() {
            if (navigator.share) {
                navigator.share({ title: 'AR City Quest', text: `I scored ${score} pts! Join the adventure!`, url: window.location.href });
            } else {
                alert(`Score: ${score} - Share this link: ${window.location.href}`);
            }
        }

        // Load saved score
        score = parseInt(localStorage.getItem('arQuestScore')) || 0;
        scoreEl.textContent = score;
    </script>
